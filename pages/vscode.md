"settings.cycle": [
{
"id": "toggleFormatOnSave",
"overrideWorkspaceSettings": true,
"values": [
{
"editor.formatOnSave": true
},
{
"editor.formatOnSave": false
}
]
}
]
}
lues": [
                  {
                      "editor.formatOnSave": true
                  },
                  {
                      "editor.formatOnSave": false
                  }
              ]
          }
      ]
  }
  ```
